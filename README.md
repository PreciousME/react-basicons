# react-croppa

React package for simple customizable icons
https://basicons.xyz

## Installation

```
npm i react-basicons
```

## Usage

```ts
import { CheckGoodYes as Check } from 'react-basicons'

const DownloadCompleteComponent: React.FC<IconConfig> = () => (
    <div>
        <Check />
        <span>Download complete!</span>
    </div>
)
```
